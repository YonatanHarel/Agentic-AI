#!/usr/bin/env python3
"""
Python Program: Calculate First 10,000 Terms of Leibniz Series
Series: 1 - 1/3 + 1/5 - 1/7 + ... (multiplied by 4)

This program calculates the first 10,000 terms of the Leibniz series,
which approximates π/4, and then multiplies the result by 4 to approximate π.

Author: Senior Python Developer
Date: 2024
"""

import time
import math

def calculate_leibniz_series_optimized(num_terms):
    """
    Calculate the Leibniz series for pi/4: 1 - 1/3 + 1/5 - 1/7 + ...
    Uses optimized approach avoiding repeated exponentiation.
    
    Args:
        num_terms (int): Number of terms to calculate
    
    Returns:
        float: Sum of the series
    """
    series_sum = 0.0
    sign = 1  # Start with positive sign
    
    for n in range(num_terms):
        denominator = 2 * n + 1
        term = sign / denominator
        series_sum += term
        sign *= -1  # Alternate the sign for next iteration
    
    return series_sum

def main():
    """Main function to execute the program"""
    print("Python Program: Calculate First 10,000 Terms of Leibniz Series")
    print("Series: 1 - 1/3 + 1/5 - 1/7 + ... (multiplied by 4)")
    print("=" * 60)
    print()
    
    # Configuration
    NUM_TERMS = 10000
    
    print(f"Calculating the first {NUM_TERMS} terms...")
    start_time = time.time()
    
    # Calculate the series
    series_sum = calculate_leibniz_series_optimized(NUM_TERMS)
    final_result = series_sum * 4
    
    end_time = time.time()
    execution_time = end_time - start_time
    
    # Display results
    print(f"\nResults:")
    print(f"Sum of series (first {NUM_TERMS} terms): {series_sum}")
    print(f"Final result (multiplied by 4): {final_result}")
    print(f"Execution time: {execution_time:.6f} seconds")
    print()
    
    # Mathematical context
    print("Mathematical Context:")
    actual_pi = math.pi
    error = abs(final_result - actual_pi)
    accuracy_percentage = (1 - error / actual_pi) * 100
    
    print(f"This series approximates π (pi)")
    print(f"Calculated value: {final_result}")
    print(f"Actual π value:  {actual_pi}")
    print(f"Absolute error:  {error}")
    print(f"Accuracy: {accuracy_percentage:.6f}%")
    print()
    
    # Show first few terms for verification
    print("First 10 terms of the series:")
    for i in range(10):
        sign = 1 if i % 2 == 0 else -1
        denominator = 2 * i + 1
        term = sign / denominator
        print(f"Term {i+1}: {sign}/{denominator} = {term:8.6f}")
    
    print(f"\nProgram completed successfully!")
    return final_result

if __name__ == "__main__":
    result = main()
